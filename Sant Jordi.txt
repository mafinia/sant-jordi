<script>

let seen = {
    carta:false,
    bolsa:false,
    libro:false
};

function checkDate() {
    let input = document.getElementById("dateInput").value;
    let selected = new Date(input);
    let limit = new Date("2026-04-23");

    if (selected < limit) {
        document.getElementById("msg").innerText =
        "Acceso restringido, vuelva a intentarlo más adelante.";
    } else {
        document.getElementById("screen1").classList.add("hidden");
        document.getElementById("screen2").classList.remove("hidden");
        startPetals();

        setTimeout(()=>{
            document.getElementById("continueBtn").classList.remove("hidden");
        },5000);
    }
}

function startPetals(){
    setInterval(()=>{
        let petal = document.createElement("div");
        petal.classList.add("petal");
        petal.style.left = Math.random()*100+"vw";
        petal.style.animationDuration = (Math.random()*3+2)+"s";
        document.body.appendChild(petal);

        setTimeout(()=>petal.remove(),5000);
    },300);
}

function goToPackage(){
    document.getElementById("screen2").classList.add("hidden");
    document.getElementById("screen3").classList.remove("hidden");
}

function shake(){
    document.getElementById("packageMsg").innerText =
    "Efectivamente, hay algo dentro.";
}

function openCarefully(){
    document.getElementById("packageMsg").innerText =
    "Abriendo con cuidado...";
    setTimeout(showContent,2000);
}

function destroy(){
    document.getElementById("packageMsg").innerHTML =
    "<div class='fight'>💥</div>";
    setTimeout(showContent,3000);
}

function showContent(){
    document.getElementById("screen3").classList.add("hidden");
    document.getElementById("screen4").classList.remove("hidden");
}

function openCarta(){
    seen.carta = true;
    showInteraction("<img src='carta_abierta.jpg'><p>💌 Carta</p>");
}

function openBolsa(){
    seen.bolsa = true;
    showInteraction("<img src='iman.jpg'><p>🎁 Imán</p>");
}

function openLibro(){
    seen.libro = true;
    showInteraction("<img src='libro_real.jpg'><p>📖 Libro</p>");
}

function showInteraction(content){
    document.getElementById("screen4").classList.add("hidden");
    document.getElementById("interaction").classList.remove("hidden");
    document.getElementById("interactionContent").innerHTML = content;
}

function nextItem(){
    document.getElementById("interaction").classList.add("hidden");

    if(seen.carta && seen.bolsa && seen.libro){
        document.getElementById("final").classList.remove("hidden");
    } else {
        document.getElementById("screen4").classList.remove("hidden");
    }
}

</script>
